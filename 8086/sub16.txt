/*************  ✨ Codeium Command ⭐  *************/
; subtraction of two 16 bit numbers


; implicit addressing mode
5000 :: MOV AX,[3000]  ; load the value at memory address 3000 into AX
        MOV BX,[4000]  ; load the value at memory address 4000 into BX
        SUB AX,BX      ; subtract the value in BX from the value in AX
        MOV [6000],AX  ; store the result from AX to memory address 6000
        HLT             ; halt the program





; explicit addressing mode
5000 :: MOV AX,0100    ; load the immediate value 0100 into AX
        MOV BX,0050    ; load the immediate value 0050 into BX
        SUB AX,BX      ; subtract the value in BX from the value in AX
        MOV [6000],AX  ; store the result from AX to memory address 6000
        HLT             ; halt the program
